* Perfil
```
create table public.loja_perfil (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  nome text null,
  email text null,
  
  user_id uuid null,
  
  foto_perfil text null,
  sexo text null,
  pais text null,
  estado text null,
  
  constraint loja_perfil_pkey primary key (id),
  constraint loja_perfil_user_id_fkey foreign KEY (user_id) references auth.users (id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;
```

* Produtos
```
create table public.loja_produto (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  nome text null,
  descricao text null,
  preco numeric null,
  imagem text null,
  
  user_id uuid null,
  
  constraint loja_produto_pkey primary key (id),
  constraint loja_produto_user_id_fkey foreign KEY (user_id) references auth.users (id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;
```
